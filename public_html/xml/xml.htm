As you can see, browsers "understand" that how an XML file should be displayed.

Now let's move on to processing XML data in JavaScript.

All of these examples should run in any browser as long as the files are on the server.

However, if you download the examples, they won't run in any browser except Firefox. All the other browsers forbid the loading of a local XML file. So if you want to test your examples on your own machine before uploading, only test in Firefox.

The next example will have an HTML file "read" the contents of the first XML file, and display it in a table. Note how an xml file gets loaded. Also note how we can access the children of a node, how we can access a particular child, and how we can obtain the text of a node. The code for this example is:

<HTML>

<HEAD>

<TITLE>Example 1</TITLE>

    <SCRIPT LANGUAGE="JavaScript">

 

     // Code to load an xml document into a variable names xmlhttp

 

          if (window.XMLHttpRequest)

          {// code for IE7+, Firefox, Chrome, Opera, Safari

              xmlhttp=new XMLHttpRequest();

          }

          else

          {// code for IE6, IE5

              xmlhttp=new ActiveXObject("Microsoft.XMLHTTP");

          }

          xmlhttp.open("GET","customers.xml",false);

          xmlhttp.send();

          xmlDoc=xmlhttp.responseXML;

 

      function start()

      {

        var table = "";

        table += "<TABLE BORDER=2>";

        table += "<TR><TH>Customer Number</TH><TH>Name</TH><TH>Address</TH>" +

                         "<TH>City</TH><TH>State</TH><TH>Zip</TH><TH>Balance</TH></TR>"; 

 

        // this loads all children of the root named Customer into an array of nodes named customers

        customers = xmlDoc.getElementsByTagName("Customer");

 

        // loop over each customer node

        for (i = 0; i < customers.length; i++)

        {

          // customer stores the ith customer node

           var customer = customers[i];

          

           // get the first (and only) CustNo child of that customer node

           // then get the node value of the first (and only) child of that CustNo node

           // which is a text node

           custno = customer.getElementsByTagName("CustNo")[0].childNodes[0].nodeValue;

 

          // and so on for the other children of the Customer node

           custfirstname = customer.getElementsByTagName("CustFirstName")[0].childNodes[0].nodeValue;

           custlastname = customer.getElementsByTagName("CustLastName")[0].childNodes[0].nodeValue;

           custstreet = customer.getElementsByTagName("CustStreet")[0].childNodes[0].nodeValue;

           custcity = customer.getElementsByTagName("CustCity")[0].childNodes[0].nodeValue;

           custstate = customer.getElementsByTagName("CustState")[0].childNodes[0].nodeValue;

           custzip = customer.getElementsByTagName("CustZip")[0].childNodes[0].nodeValue;

           custbal = customer.getElementsByTagName("CustBal")[0].childNodes[0].nodeValue;

 

 

           table += "<TR>";

           table += "<TD>"+ custno + "</TD>";

            table += "<TD>"+ custfirstname + " " + custlastname + "</TD>";

           table += "<TD>"+ custstreet + "</TD>";

           table += "<TD>"+ custcity + "</TD>";

           table += "<TD>"+ custstate + "</TD>";

           table += "<TD>"+ custzip + "</TD>";

           table += "<TD>"+ custbal + "</TD>";         

           table += "</TR>";

         }

         table += "</TABLE>";

         document.getElementById("display").innerHTML = table;

       }

    </SCRIPT>

</HEAD>

<BODY onload="start()">

     <div id="display"></div>

</BODY>

</HTML>

Well see that the count is different in Firefox (and Chrome) than it is in Internet Explorer.

 

In Firefox, there are 33 children; in Internet Explorer, there are 16.

 

Internet Explorer ignores white space between XML nodes.

 

Firefox and Chrome think put that white space in text nodes.

In this example, well create the headings for the table from the first child of the root. And then add the rows of data. ex3.htm

 

<HTML>

<HEAD>

<TITLE>Example 3</TITLE>

    <SCRIPT LANGUAGE="JavaScript">

   

          if (window.XMLHttpRequest)

          {// code for IE7+, Firefox, Chrome, Opera, Safari

              xmlhttp=new XMLHttpRequest();

          }

          else

          {// code for IE6, IE5

              xmlhttp=new ActiveXObject("Microsoft.XMLHTTP");

          }

          xmlhttp.open("GET","customers.xml",false);

          xmlhttp.send();

          xmlDoc=xmlhttp.responseXML;

         

      function start()

      {

        var table = "";

        childrenNodes = xmlDoc.childNodes[0].childNodes;

        // xmlDoc.childNodes is an array that consists of the root element

        // xmlDoc.childNodes[0].childNodes is an array of the root's children

       

        // test to see if first child is an element (nodeType of 1)

        if (childrenNodes[0].nodeType == 1)

        {

          heading = childrenNodes[0];

        }

        else // second child must be the first element

        {

          heading = childrenNodes[1];

        }

         

         table += "<TABLE BORDER=2>";

         table += "<TR>";

      

         // heading is first element child of the root

          for (j = 0; j < heading.childNodes.length; j++)

          {

           if (heading.childNodes[j].nodeType == 1) // if node is an element

           {

               // display the name of the node in a heading

                   table += "<TH>" + heading.childNodes[j].nodeName + "</TH>";

            }

          }

          table += "</TR>";

 

         // add the data

        for (i = 0; i < childrenNodes.length; i++)

        {

           // get a child of the root

           thisChild = childrenNodes[i];

           if (thisChild.nodeType == 1) // if it's an element

           {

                   table += "<TR>";

                   // for each child of that node

                   for (j = 0; j < thisChild.childNodes.length; j++)

                   {

                        // get a child of that node

                        thatProperty = thisChild.childNodes[j];

                        if (thatProperty.nodeType == 1)       

                        {

                             // get the value of the first child of that node (the text)

                            table += "<TD>" + thatProperty.childNodes[0].nodeValue + "</TD>";

                        }

                   }

            }

            table += "</TR>";

         }

         table += "</TABLE>";

         document.getElementById("display").innerHTML = table;

       }

    </SCRIPT>

 

</HEAD>

<BODY onload="start()">

     <div id="display"></div>

</BODY>

</HTML>

          // heading is first element child of the root

          // new code

              attributesArray = heading.attributes;

              for (var j = 0; j < attributesArray.length; j++)

              {

                   table += "<TH>"+ attributesArray[j].name + "</TH>";

              }

          // end of new code

        for (j = 0; j < heading.childNodes.length; j++)

        {

              if (heading.childNodes[j].nodeType == 1) // if node is an element

              {

                   // display the name of the node in a heading

                   table += "<TH>" + heading.childNodes[j].nodeName + "</TH>";

              }

        }

        table += "</TR>";

 

              // add the data

        for (i = 0; i < childrenNodes.length; i++)

        {

           // get a child of the root

           thisChild = childrenNodes[i];

           if (thisChild.nodeType == 1) // if it's an element

          {

                   table += "<TR>";

                   // new code

                        attributesArray = thisChild.attributes;

                        for (var j = 0; j < attributesArray.length; j++)

                        {

                             table += "<TD>"+ attributesArray[j].value + "</TD>";

                        }

                   // end of new code

                   // for each child of that node

                        for (j = 0; j < thisChild.childNodes.length; j++)

                        {

                            // get a child of that node

                            thatProperty = thisChild.childNodes[j];

                            if (thatProperty.nodeType == 1)       

                             {

                                  // get the value of the first child of that node (the text)

                                 table += "<TD>" + thatProperty.childNodes[0].nodeValue + "</TD>";

                             }

                        }

           }

          table += "</TR>";

         }

         table += "</TABLE>";
